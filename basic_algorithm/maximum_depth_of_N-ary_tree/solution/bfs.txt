class Solution {
    public int maxDepth(Node root) {
        if (null == root) {
            return 0;
        }
        Queue<Node> queue = new LinkedList<Node>();
        queue.offer(root);
        int size = queue.size();
        int depth = 0;
        
        while (!queue.isEmpty()) {
            Node curNode = queue.poll();
            for (Node node : curNode.children) {
                if (null != node) {
                   queue.offer(node); 
                }
            }
            if (0 == --size) {
                size = queue.size();
                depth++;
            }
        }
        
        return depth;
    }
}
